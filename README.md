void gotoxy(int x, int y)
{
	COORD Cur;

	Cur.Y = y;

}

void move()
{
	register int kb;

	kb = getch();

	switch (kb)
	{
	case LEFT:
		if (map[0][cursor1.cy][cursor1.cx / 2 - 1] == 0)
		{
			gotoxy(cursor1.cx, cursor1.cy);
			BLUE printf("  ");
			cursor1.cx -= 2;
			gotoxy(cursor1.cx, cursor1.cy);
			BLUE printf("%s", player);
			gotoxy(48, 26);
		}
		break;

	case RIGHT:
		if (map[0][cursor1.cy][cursor1.cx / 2 + 1] == 0)
		{
			gotoxy(cursor1.cx, cursor1.cy);
			BLUE printf("  ");
			cursor1.cx += 2;

			BLUE printf("%s", player);
			gotoxy(48, 26);
		}
		break;
	}
}
void move_menu()
{
	int kb1 = getch();
	switch (kb1)
	{
	case DOWN:
		if (cursor1.by + 3 <= 23)
		{
			gotoxy(16, cursor1.by);
			printf("  ");
			gotoxy(32, cursor1.by);
			printf("  ");
			cursor1.by += 3;
			gotoxy(16, cursor1.by);
			printf("▷");
			gotoxy(32, cursor1.by);
			printf("◁");
		}
		break;
	case UP:
		if (cursor1.by - 3 >= 17)
		{
			gotoxy(16, cursor1.by);
			printf("  ");
			gotoxy(32, cursor1.by);
			printf("  ");
			cursor1.by -= 3;
			gotoxy(16, cursor1.by);
			printf("▷");
			gotoxy(32, cursor1.by);
			printf("◁");
		}
		break;
	case ENTER:
		if (game_source1.menu_change == 0 && cursor1.by == 17)
			game_source1.begin_menu = 1;
		else if (game_source1.menu_change == 0 && cursor1.by == 20)
		{
			//system("cls");
			game_source1.menu_change = 1;
		}
		else if (game_source1.menu_change == 0 && cursor1.by == 23)
			exit(0);

		break;
	}
}
